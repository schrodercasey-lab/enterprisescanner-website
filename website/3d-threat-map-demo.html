Implementation Guide

1. Include Dependencies

<!-- Three.js -->
<script src="https://cdn.jsdelivr.net/npm/three@0.158.0/build/three.min.js"></script>

<!-- Component CSS -->
<link rel="stylesheet" href="css/3d-threat-map.css">

<!-- Component JS -->
<script src="js/3d-threat-map.js"></script>

2. HTML Structure

<div class="threat-map-wrapper">
    <div id="threat-map-3d"></div>
</div>

3. JavaScript Initialization

// Auto-initializes on DOMContentLoaded
// Or manually create instance:

const threatMap = new ThreatMap3D('threat-map-3d', {
    autoRotate: true,
    autoRotateSpeed: 0.2,
    maxThreats: 50,
    enableZoom: true,
    minZoom: 200,
    maxZoom: 500
});

4. Configuration Options

{
    globeRadius: 100,           // Globe size
    cameraDistance: 300,       // Initial camera position
    autoRotate: true,           // Enable auto-rotation
    autoRotateSpeed: 0.2,       // Rotation speed
    enableZoom: true,           // Allow zoom
    minZoom: 200,              // Min zoom distance
    maxZoom: 500,              // Max zoom distance
    threatPulseSpeed: 0.03,     // Marker pulse animation
    dataFlowSpeed: 0.02,        // Data flow animation
    maxThreats: 50              // Max concurrent threats
}
